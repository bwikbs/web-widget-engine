<test name='namednodemapreturnnull' schemaLocation='http://www.w3.org/2001/DOM-Test-Suite/Level-1 dom1.xsd'>
     <metadata>
          <title>namednodemapReturnNull</title>
          <creator>NIST</creator>
          <description> The "getNamedItem(name)" method returns null of the specified name did not identify any node in the map. Retrieve the second employee and create a NamedNodeMap listing of the attributes of the last child. Once the list is created an invocation of the "getNamedItem(name)" method is done with name="district". This name does not match any names in the list therefore the method should return null. </description>
          <contributor>Mary Brady</contributor>
          <date qualifier='created'>2001-08-17</date>
          <subject resource='http://www.w3.org/TR/1998/REC-DOM-Level-1-19981001/level-one-core#ID-1074577549'/>
</metadata>

     <var name='elementList' type='NodeList'/>

     <var name='attributes' type='NamedNodeMap'/>
     <var name='districtNode' type='Attr'/>
     <load var='doc' href='staff' willBeModified='false'/>
     <getElementsByTagName interface='Document' obj='doc' var='elementList' tagname='"address"'/>
     <item interface='NodeList' obj='elementList' var='testEmployee' index='1'/>
