<test name='nodeinsertbeforenodeancestor' schemaLocation='http://www.w3.org/2001/DOM-Test-Suite/Level-1 dom1.xsd'>
     <metadata>
          <title>nodeInsertBeforeNodeAncestor</title>
          <creator>NIST</creator>
          <description> The "insertBefore(newChild,refChild)" method raises a HIERARCHY_REQUEST_ERR DOMException if the node to be inserted is one of this nodes ancestors. Retrieve the second employee and attempt to insert a node that is one of its ancestors(root node). An attempt to insert such a node should raise the desired exception. </description>
          <contributor>Mary Brady</contributor>
          <date qualifier='created'>2001-08-17</date>

          <subject resource='http://www.w3.org/TR/1998/REC-DOM-Level-1-19981001/level-one-core#ID-952280727'/>
          <subject resource='http://www.w3.org/TR/1998/REC-DOM-Level-1-19981001/level-one-core#xpointer(id('ID-952280727')/raises/exception[@name='DOMException']/descr/p[substring-before(.,':')='HIERARCHY_REQUEST_ERR'])'/>
          <subject resource='http://www.w3.org/TR/1998/REC-DOM-Level-1-19981001/level-one-core#ID-952280727'/>

</metadata>
