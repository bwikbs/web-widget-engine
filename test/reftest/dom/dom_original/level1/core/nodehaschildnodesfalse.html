          <title>nodeHasChildNodesFalse</title>

          <description> The "hasChildNodes()" method returns false if the node does not have any children. Retrieve the Text node inside the first child of the second employee and invoke the "hasChildNodes()" method. It should return the boolean value "false". </description>

</metadata>
     <var name='doc' type='Document'/>
     <var name='elementList' type='NodeList'/>
     <var name='child' type='Node'/>
     <var name='employeeIdList' type='NodeList'/>
     <var name='employeeNode' type='Node'/>
     <var name='textNode' type='Node'/>
     <var name='state' type='boolean'/>
